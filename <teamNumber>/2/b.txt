CREATE VIEW ENTITY1 AS SELECT E.WORD2 AS WORD1,E.DOC_ID AS DOC_ID FROM ENTITY E WHERE E.WORD1 = 'PIZZA';
CREATE VIEW ENTITY2 AS SELECT E.WORD2 AS WORD1,E1.DOC_ID AS DOC_ID FROM ENTITY E INNER JOIN ENTITY1 E1 ON E1.WORD1=E.WORD1 WHERE E.DOC_ID= E1.DOC_ID;
CREATE VIEW ENTITY3 AS SELECT E.WORD2 AS WORD1,E2.DOC_ID AS DOC_ID FROM ENTITY E INNER JOIN ENTITY2 E2 ON E2.WORD1=E.WORD1 WHERE E.DOC_ID= E2.DOC_ID;
SELECT WORD1,DOC_ID FROM ENTITY3;
